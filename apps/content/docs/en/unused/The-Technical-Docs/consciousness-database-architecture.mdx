### Analytics and Reporting

#### Session Analytics

```sql
-- Aggregated session analytics
CREATE MATERIALIZED VIEW session_analytics AS
SELECT 
  s.id,
  s.technique_id,
  s.duration_seconds,
  s.coherence_score,
  COUNT(DISTINCT cs.id) as stream_count,
  AVG(cs.heart_rate_variability) as avg_hrv,
  ARRAY_AGG(DISTINCT cs.active_sphere_nodes) as unique_nodes,
  COUNT(ld.id) as loop_detections,
  MAX(qc.coherence_level) as peak_coherence
FROM breath_sessions s
LEFT JOIN consciousness_streams cs ON cs.session_id = s.id
LEFT JOIN loop_detection ld ON ld.stream_id = cs.id
LEFT JOIN quantum_coherence qc ON qc.session_id = s.id
GROUP BY s.id, s.technique_id, s.duration_seconds, s.coherence_score;

-- Refresh every hour
CREATE INDEX idx_session_analytics_refresh ON session_analytics(id);
```

#### Protection Effectiveness

```sql
-- Track overall protection system effectiveness
CREATE VIEW protection_effectiveness AS
SELECT 
  DATE_TRUNC('day', created_at) as day,
  COUNT(*) as total_detections,
  SUM(CASE WHEN mitigation_applied THEN 1 ELSE 0 END) as mitigations,
  SUM(CASE WHEN null_state_triggered THEN 1 ELSE 0 END) as null_states,
  AVG(severity) as avg_severity,
  COUNT(DISTINCT stream_id) as affected_sessions
FROM loop_detection
GROUP BY DATE_TRUNC('day', created_at)
ORDER BY day DESC;
```

